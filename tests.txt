from-string-simple
  extensions [ csv ]
  csv:from-string "1,2,3" => ["1" "2" "3"]

from-string-quotes
  extensions [ csv ]
  csv:from-string "\"1\",2,\"3,4\"" => ["1" "2" "3,4"]

from-string-nested-quotes
  extensions [ csv ]
  csv:from-string "\"\"\"quoted string\"\"\",\"inner\"\"quote\",\"escaped\\\"\"quote\"" => ["\"quoted string\"" "inner\"quote" "escaped\\\"quote"]
