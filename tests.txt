to-strings-simple
  extensions [ csv ]
  csv:to-strings "1,2,3" => ["1" "2" "3"]

to-strings-quotes
  extensions [ csv ]
  csv:to-strings "\"1\",2,\"3,4\"" => ["1" "2" "3,4"]

to-strings-nested-quotes
  extensions [ csv ]
  csv:to-strings "\"\"\"quoted string\"\"\",\"inner\"\"quote\",\"escaped\\\"\"quote\"" => ["\"quoted string\"" "inner\"quote" "escaped\\\"quote"]

to-strings-and-numbers
  extensions [ csv ]
  csv:to-strings-and-numbers "1,one,2,\"3\",pi,e" => [1 "one" 2 3 "pi" "e"]

to-strings-custom-delimiter
  extensions [ csv ]
  (csv:to-strings "1\t2\t3" "\t") => ["1" "2" "3"]
  (csv:to-strings "1;2,3" ";") => ["1" "2,3"]

from-list-to-string
  extensions [ csv ]
  csv:from-list-to-string [1 2 3] => "1,2,3"
  csv:from-list-to-string ["one" "two" "three"] => "one,two,three"
  csv:from-list-to-string [1 "two" "three,four"] => "1,two,\"three,four\""
  (csv:from-list-to-string [1 2 3] "\t") => "1\t2\t3"

from-list-to-readable-strings
  extensions [ csv ]
  csv:from-list-to-readable-string [1 2 3] => "1,2,3"
  csv:from-list-to-readable-string ["one" "two" "three"] => "\"\"\"one\"\"\",\"\"\"two\"\"\",\"\"\"three\"\"\""
